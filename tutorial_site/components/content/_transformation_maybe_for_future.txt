const highlighter = await getHighlighter({
  themes: Object.keys(bundledThemes),
  langs: Object.keys(bundledLanguages),
})


async function transformContentAsync(rawContent: string): Promise<string> {
  return highlighter.codeToHtml(rawContent, {
    lang: props.language,
    theme: 'material-theme-darker',
  })
}

const rawContent = props.code; // Initial raw content
const transformedContent = ref<string>(''); // HTML content after transformation

onMounted(async () => {
  transformedContent.value = await transformContentAsync(rawContent.valueOf());
});
